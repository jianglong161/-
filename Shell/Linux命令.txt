echo:终端打印命令
	echo >
	echo >>
pgrep:获得进程
var：赋值
alias:使用别名 ，简化命令
tput：获取终端信息
find:查找
	find . /print :打印出匹配文件
tr：转换
md5sum：进行校验
sort 排序 
unique 消除重复的文件
mktemp：生成一个临时文件并返回其文件名
spilt:分割
comm：命令可用于两个文件之间的比较
	-1：删除第一列
	-2：删除第二列
touch：建立文件
xargs: 传递参数
chmod：设置文件权限
chown:	更改用户所有权
chattr:将文件设置为不可修改
diff:生成差异文件
patch：修补文件
head：打印文件的前10行
tail：打印文件的后10行
pushd和popd：用于在多个目录之间进行切换而无需复制并粘贴目录路径
dirs：查看栈内容
wc：统计文件的个数
tree：打印目录
grep:用于文本搜索
	默认带有正则表达式命令 egrep
	grep -v ：-v将匹配的结果进行反转
		 -c: 文本中包含的字符数量
		 -e：匹配多个样式
	搜素递归 grep "text" . -R -n
cut：将文本按列进行切分的小巧工具
	-b 用于表示字节
	-c 表示字符
	-f 用于自定义字段
sed:文本替换
	如果替换所有内容，在尾部换成g
awk: 'BEGIN{ print "start"} pattern {commands} END {prinit "end"}' file
	NR：表示记录是数量，在执行的过程中对应当前行号
	NF:表示字段数量，在执行的过程中对应于当前行的字段数
	$0:这个变量包含执行过程中当前行的文本内容
	$1：这个变量包含第一个文本的内容。
seq:从1到10	
paste：合并连接
wget:用于文件下载的命令行工具
